import imaplib
import email
import datetime
import traceback
from email.header import decode_header
from email.utils import parsedate_to_datetime


class EmailRecv:
    def __init__(self, email_addr, auth_token, imap_server='imap.qq.com', imap_port=993):
        self.email_addr = email_addr
        self.auth_token = auth_token

        self.imap_server = imap_server
        self.imap_port = imap_port

        self.conn = None

    def login(self):
        if self.conn is None:
            username, password, imap_server, imap_port = self.email_addr, self.auth_token, self.imap_server, self.imap_port
            conn = imaplib.IMAP4_SSL(imap_server, imap_port)
            conn.login(username, password)
            self.conn = conn
        return self.conn

    @classmethod
    def decode_mime_str(cls, s):
        """解码邮件头（标题、发件人等）"""
        if not s: return ""
        decode_list = decode_header(s)
        result = ""
        for content, charset in decode_list:
            if isinstance(content, bytes):
                result += content.decode(charset if charset else "utf-8", errors="ignore")
            else:
                result += content
        return result

    @classmethod
    def get_body(cls, msg):
        """解析邮件正文"""
        if msg.is_multipart():
            # 如果邮件是多部分组成的（文本+附件+HTML）
            for part in msg.walk():
                content_type = part.get_content_type()
                content_disposition = str(part.get("Content-Disposition"))
                
                # 只要正文，不要附件
                if content_type == "text/plain" and "attachment" not in content_disposition:
                    charset = part.get_content_charset() or 'utf-8'
                    return part.get_payload(decode=True).decode(charset, errors='ignore')
        else:
            # 非多部分邮件
            charset = msg.get_content_charset() or 'utf-8'
            return msg.get_payload(decode=True).decode(charset, errors='ignore')
        return ""

    def fetch(self, unseen=True, from_addr=None, since_date=None):
        # 1. 登录
        conn = self.login()
        conn.select("INBOX")

        # 2. 搜索邮件
        condition = 'UNSEEN' if unseen else 'ALL'
        if from_addr:
            condition += f' FROM "{from_addr}"'
        if since_date:
            condition += f' SINCE "{since_date}"'
        # print(f'{condition}')

        status, messages = conn.search(None, condition)  # UNSEEN ALL  FROM "noreply@tavily.com" SINCE "25-Feb-2026"
        if status != "OK" or not messages[0]:
            return list()

        # 3. 获取邮件
        results = list()
        mail_ids = messages[0].split(); # print(mail_ids)
        for m_id in reversed(mail_ids):
            # 获取邮件header和body
            res, data = conn.fetch(m_id, "(RFC822)")  # RFC822.HEADER 可以仅抓取邮件头信息（Date, Subject, From），节省流量
            msg = email.message_from_bytes(data[0][1])

            # 解析header和body
            mail_date = parsedate_to_datetime(msg.get("Date"))
            subject = self.decode_mime_str(msg.get("Subject"))
            sender = self.decode_mime_str(msg.get("From"))
            body = self.get_body(msg)

            results.append({"date": mail_date.astimezone().strftime('%Y-%m-%d %H:%M:%S'), "subject": subject, "sender": sender, "body": body})
        return sorted(results, key=lambda x: x['date'])
